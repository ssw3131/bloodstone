bloodstone은 jquery 기반 위에서 작동됩니다.

					dk.INFO;
				

				
			

		

		

		

			

			

				
개발 작업 시 debug 모드를 활성화 해서 개발합니다. isDebug 가 true 일 경우에만 log 함수가 작동 됩니다.

				
log 함수는 console.log를 대체합니다.
log 함수에 첫번째 파라미터가 'debug' 일 경우 debugger; 중단점이 실행됩니다.

				
					dk.isDebug( true );
					log( 'hi' );
					log( 'debug', 'hi' );
				

			

		

		

		

			

			

				
Detector를 통해서 데스크탑, 모바일, 브라우저, 객체 탐지 등을 미리 파악할 수 있습니다.

				
					dk.DETECTOR;
				

				

				
				
					// ie 일 경우
					if( dk.DECTOER.browser == 'ie' ){
					} else {
					}
	     

			

		

		

		

			

			

				
유틸리티 함수들


				

					
random

					
특정범위 내에 랜덤숫자를 반환합니다.
max 기본값 : 1, min 기본값 : 0

					
						// min ~ max
						dk.random( max, min );

						// 0 ~ max
						dk.random( max );

						// 0 ~ 1
						dk.random();
					

				

				
				

					
randomInt

					
특정범위 내에 랜덤정수를 반환합니다.
min 기본값 : 0

					
						// min ~ max
						dk.randomInt( max, min );

						// 0 ~ max
						dk.randomInt( max );
					

				

				
				

					
randomColor

					
랜덤 컬러값을 반환합니다.

					
						dk.randomColor();
					

				

				
				

					
timeCheck

					
timeCheck 함수 실행 사이의 시간을 반환합니다.

					
						dk.timeCheck();
						// ...
						dk.timeCheck();
					

				

				
				

					
parseLocation

					
location.search의 문자열을 & 문자열로 나눠 오브젝트에 key, value 값으로 저장해 반환합니다.

					
						// index.html?a=1&b=2;
						dk.parseLocation();
						// {a: "1", b: "2"}
					

				

				
			

		

		

		

			

			

				
리스트의 length 값을 받아 리스트 안의 활성화 id 값을 관리합니다.

				
활성화 id 값을 바꾸고 현재 id와 이전 id 값을 제공합니다. autoPlay 기능을 제공합니다.

				
배열의 id, 슬라이드의 id, 버튼리스트의 id 등에 활용 가능합니다.

				
					/**
					* @class	: ListManager
					* @param	: leng - list length
					* @param	: act - 활성화 함수
					* @param	: option - initId, infinity, autoPlay, autoPlaySpeed
					*/
					dk.ListManager( leng, act, option );


				

				
					dk.ListManager( 4, function( id, oldId, directon ){}, {
						initId: 0,
						infinity: true,
						autoPlay: false,
						autoPlaySpeed: 4
					} );

				

				

					
leng

					
Type : int

					
leng 은 필수항목 입니다.

					
leng 이 없거나 0 이면 안됩니다.

				

				
				

					
act

					
Type : function

					
act 는 필수항목 입니다.

					
활성화 id 값이 바뀔때 실행될 리스너 함수입니다.

				

				
				

					
Option

					

						
initId

						
Type : int

						
초기 설정 될 id 값

					

					

						
infinity

						
Type : boolean

						
Default : true

						
autoPlay 시 무한반복 설정

					

					

						
autoPlay

						
Type : boolean

						
Default : false

						
autoPlay 설정

					

					

						
autoPlaySpeed

						
Type : number

						
Default : 4

						
autoPlay 시간 설정 (초)

					

				

				
				

					
Method

					

						
destroy()

						
autoPlay 및 모든 기능이 제거됩니다.

						
							var lm = dk.ListManager( leng, act, option );
							lm.destroy();
						

					

					

						
getId()

						
return : { id, oldId }

						
id (현재 활성화 id)와 oldId (이전 id) 를 반환합니다.

						
							var lm = dk.ListManager( leng, act, option );
							lm.getId();
							// { id: 1, oldId: 0 }
						

					

					

						
act( id )

						
act( id, [directon] )

						
Arguments - id:int

						
Arguments - directon:string ('next' or 'prev') / 선택적

						
현재 활성화 id를 지정합니다.

						
이전 id 대비 이동 방향을 고정 시킬 수 있습니다. 'next' or 'prev'

						
							var lm = dk.ListManager( leng, act, option );
							lm.act( 2 );
							lm.act( 1, 'next' );
						

					

					

						
next()

						
다음 id로 이동

						
							var lm = dk.ListManager( leng, act, option );
							lm.next();
						

					

					

						
prev()

						
이전 id로 이동

						
							var lm = dk.ListManager( leng, act, option );
							lm.prev();
						

					

					

						
play()

						
autoPlay를 실행시킵니다.

						
							var lm = dk.ListManager( leng, act, option );
							lm.play();
						

					

					

						
stop()

						
autoPlay를 멈춥니다.

						
resetTimer를 실행해도 재실행하지 않습니다.

						
							var lm = dk.ListManager( leng, act, option );
							lm.stop();
						

					

					

						
pause()

						
autoPlay를 일시정지 합니다.

						
resetTimer를 실행하면 재실행합니다.

						
							var lm = dk.ListManager( leng, act, option );
							lm.pause();
						

					

					

						
resetTimer()

						
autoPlay를 재실행 시킵니다.

						
							var lm = dk.ListManager( leng, act, option );
							lm.resetTimer();
